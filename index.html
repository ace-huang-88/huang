import time import json import requests import threading from datetime import datetime from plyer import notification # 桌面弹窗依赖 # ============== 配置区（可自行修改）============== ALERT_5MIN_RISE = 5 # 5分钟涨幅阈值（%） ALERT_10MIN_RISE = 10 # 10分钟涨幅阈值（%） CHECK_INTERVAL = 10 # 每10秒抓取一次数据（建议不低于5秒） HISTORY_FILE = "coinglass_history.json" # 历史数据存储文件 # ================================================= # 全局存储：币种历史数据 coin_history = {} headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36", "Referer": "https://www.coinglass.com/" } def load_history(): """加载本地历史数据""" global coin_history try: with open(HISTORY_FILE, "r", encoding="utf-8") as f: coin_history = json.load(f) except: coin_history = {} def save_history(): """保存历史数据到本地""" with open(HISTORY_FILE, "w", encoding="utf-8") as f: json.dump(coin_history, f, ensure_ascii=False, indent=2) def get_all_coins_data(): """从CoinGlass API获取所有币种持仓数据""" url = "https://api.coinglass.com/v2/public/open_interest/list" try: resp = requests.get(url, headers=headers, timeout=10) if resp.status_code == 200: return resp.json().get("data", []) except Exception as e: print(f"数据获取失败: {str(e)}") return [] def calculate_rise(now, past): """计算涨幅百分比""" if past <= 0: return 0 return round(((now - past) / past) * 100, 2) def check_alert(symbol, now_val, now_time): """检查是否触发预警""" history = coin_history.get(symbol, []) if len(history) < 2: return # 遍历历史数据，找5分钟/10分钟前的记录 for record in history: t = record["time"] val = record["value"] sec_diff = now_time - t # 5分钟预警（±10秒误差） if 290 <= sec_diff <= 310: rise = calculate_rise(now_val, val) if rise > ALERT_5MIN_RISE: send_alert(symbol, rise, "5分钟", ALERT_5MIN_RISE) # 10分钟预警（±10秒误差） if 590 <= sec_diff <= 610: rise = calculate_rise(now_val, val) if rise > ALERT_10MIN_RISE: send_alert(symbol, rise, "10分钟", ALERT_10MIN_RISE) def send_alert(symbol, rise, time_type, threshold): """发送桌面弹窗预警""" try: notification.notify( title=f"🚨 持仓量暴涨预警 {symbol}", message=f"{time_type}内上涨 {rise}% (阈值>{threshold}%)\n时间：{datetime.now().strftime('%H:%M:%S')}", timeout=10 ) print(f"【预警】{symbol} | {time_type}上涨 {rise}%") except: print(f"【预警】{symbol} 持仓量暴涨 {rise}%") def monitor(): """主监控循环""" print("="*50) print("CoinGlass 全币种持仓监控已启动") print(f"预警规则：5分钟>{ALERT_5MIN_RISE}% | 10分钟>{ALERT_10MIN_RISE}%") print("="*50) load_history() while True: now_ts = int(time.time()) coins = get_all_coins_data() if not coins: time.sleep(CHECK_INTERVAL) continue # 处理每个币种 for coin in coins: symbol = coin.get("symbol", "") open_interest = coin.get("openInterest", 0) if not symbol or open_interest <= 0: continue # 检查预警 check_alert(symbol, open_interest, now_ts) # 更新历史记录（只保留最近20分钟数据，避免过大） if symbol not in coin_history: coin_history[symbol] = [] coin_history[symbol].append({ "time": now_ts, "value": open_interest }) # 保留最近20分钟记录 coin_history[symbol] = [ r for r in coin_history[symbol] if now_ts - r["time"] <= 1200 ] # 定时保存 save_history() time.sleep(CHECK_INTERVAL) if __name__ == "__main__": # 安装依赖：pip install requests plyer try: monitor() except KeyboardInterrupt: print("\n监控已停止")
